142 inline void _get_current_line(
143  const char * const text,
144  size_t text_length_byte,
145  size_t current_pos,
146  char line_terminator,
147  /* out */
148  const char ** line,
149  size_t * line_length_byte)
150 {
151  ASSERT(text);
152  ASSERT(text_length_byte >= 1);
153  ASSERT(0 <= current_pos);
154  ASSERT(current_pos <= text_length_byte - 1);
155 
156  // \n aaaaaaaaaaaa \n
157  // ^ ^
158  // start end
159  const char *line_start = text + current_pos, *line_end = text + current_pos;
160 
161  // search line_start
162  while (line_start > text && *(line_start - 1) != line_terminator) --line_start;
163  // search line_end
164  while (line_end < text + text_length_byte && *line_end != line_terminator) ++line_end;
165 
166  *line = line_start;
167  *line_length_byte = line_end - line_start;
168 }

170 inline std::vector<std::string> _split(const char * const str, size_t len, char delimiter) {
171  ASSERT(str);
172  ASSERT(len >= 0);
173 
174  std::vector<std::string> ret; // NRVO optimization may prevent copy when returning this local variable.
175 
176  const char *p_beg = str, *p_end = str;
177  while (p_end - str < len) {
178  // come to delimiter
179  if (*p_end == delimiter) {
180  ret.push_back(std::string(p_beg, p_end - p_beg));
181  p_beg = p_end + 1;
182  }
183  ++p_end;
184  }
185  // come to end of str
186  ret.push_back(std::string(p_beg, p_end - p_beg));
187  return ret;
