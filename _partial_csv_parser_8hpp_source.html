108  char line_terminator = '\n')
109 {
110  ASSERT(text);

112  ASSERT(0 <= current_pos && current_pos <= text_length_byte - 1);
113 
114  // \n aaaaaaaaaaaa \n
115  // ^ ^
116  // start end
117  const char *line_start = text + current_pos, *line_end = text + current_pos;
118 
119  // search line_start
120  while (line_start > text && *(line_start - 1) != line_terminator) --line_start;
121  // search line_end
122  while (line_end < text + text_length_byte && *line_end != line_terminator) ++line_end;
123 
124  *line = line_start;
